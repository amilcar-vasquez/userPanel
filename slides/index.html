// internal/github/rank.go
func CalculateRank(stats *UserProfileStats) RankInfo {
    score := (stats.TotalCommits * 2) +
             (stats.TotalPullRequests * 3) +
             (stats.TotalIssues * 1) +
             (stats.TotalReviews * 2) +
             (stats.TotalStarsEarned * 4) +
             (stats.Followers * 1)

    rank, nextRank, threshold := determineRank(score)
    progress := calculateProgress(score, threshold)

    return RankInfo{
        Rank:               rank,
        Score:              score,
        NextRank:           nextRank,
        NextRankThreshold:  threshold,
        ProgressPercent:    progress,
    }
}

language Frontend: API Integration

// lib/github.ts
export interface GitHubProfileResponse {
    profile: GitHubProfileStats;
    rank: RankInfo;
}

export async function fetchGithubProfile(): 
    Promise<GitHubProfileResponse> {
    const response = await get<GitHubProfileResponse>(
        '/github/profile'
    );
    if (response.success && response.data) {
        return response.data;
    }
    throw new Error(response.message || 'Failed to fetch profile');
}

-- PostgreSQL Schema
CREATE TABLE users (
    id SERIAL PRIMARY KEY,
    name TEXT NOT NULL,
    email TEXT UNIQUE NOT NULL,
    password_hash TEXT NOT NULL,
    github_username TEXT,
    github_token TEXT,
    created_at TIMESTAMP NOT NULL DEFAULT NOW(),
    updated_at TIMESTAMP NOT NULL DEFAULT NOW(),
    deleted_at TIMESTAMP
);

CREATE INDEX idx_users_email ON users(email);
CREATE INDEX idx_users_deleted_at ON users(deleted_at);
